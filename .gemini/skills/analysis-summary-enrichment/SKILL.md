# SKILL: analysis-summary-enrichment

## Description
This skill extends the functionality of the `analysis-nwb-data-availability-report` by appending a comprehensive table of array shapes to the generated `DATA_AVAILABILITY_SUMMARY.md` file. For each session, it attempts to identify and load representative `.npy` files for behavioral data, Local Field Potentials (LFP), and neural unit spikes across different probes. It then extracts and formats their NumPy array dimensions (e.g., `(Trials, Channels, Samples)` or `(Trials, Units, Samples)`), providing a crucial overview of the data structure. This detailed shape information is invaluable for verifying data integrity, debugging processing pipelines, and planning subsequent analyses that rely on specific array dimensions.

## Core Tasks
1.  **Identify Sessions**: Scans the `data/nwb/` directory to identify all unique session IDs based on NWB filenames.
2.  **Retrieve Array Shapes**: For each session, it attempts to load specific `.npy` files from `data/arrays/` (behavioral, LFP for probes 0-2, unit spikes for probes 0-2 for a representative condition, e.g., "AAAB") and extracts their `shape` attributes.
3.  **Format Shape Data**: Organizes the retrieved shape information into a structured Pandas DataFrame.
4.  **Update Summary Markdown**: Reads the existing `DATA_AVAILABILITY_SUMMARY.md` file, appends a new Markdown table containing the array shape details under a dedicated heading ("Detailed Array Shapes"), and saves the updated file.

## Inputs
*   **`DATA_AVAILABILITY_SUMMARY.md`**: An existing Markdown file (expected in the root directory), typically generated by `analysis-nwb-data-availability-report`.
*   **NWB Files**: `.nwb` files in `data/nwb/` (used to identify session IDs).
*   **NumPy Data Files**: `.npy` files in `data/arrays/` structured as `ses<ID>-behavioral-<COND>.npy`, `ses<ID>-probe<P>-lfp-<COND>.npy`, and `ses<ID>-units-probe<P>-spk-<COND>.npy`.

## Outputs
*   **Updated `DATA_AVAILABILITY_SUMMARY.md`**: The original summary file is modified to include a new section with a Markdown table detailing the shapes of various behavioral, LFP, and unit spike data arrays for each session and probe.

```python
import os
import numpy as np
import pandas as pd
import re
import glob # For mocking glob.glob

# --- Mocking necessary functions and file structures ---
# Mock NWB_DIR and DATA_DIR for demonstration purposes
MOCK_NWB_DIR = 'mock_workspace'
MOCK_DATA_DIR = os.path.join(MOCK_NWB_DIR, 'data')
MOCK_NWB_PATH = os.path.join(MOCK_DATA_DIR, 'nwb')
MOCK_ARRAYS_PATH = os.path.join(MOCK_DATA_DIR, 'arrays')
MOCK_CHECKPOINTS_PATH = os.path.join(MOCK_NWB_DIR, 'checkpoints') # Used in other examples for output

def mock_get_npy_shape(path):
    # Simulate shape retrieval for mock files
    if "behavioral" in path:
        return (100, 4, 10000) # Example: (Trials, behavioral_vars, samples)
    elif "lfp" in path:
        return (100, 128, 10000) # Example: (Trials, channels, samples)
    elif "spk" in path:
        return (100, 50, 10000) # Example: (Trials, units, samples)
    return None

def mock_enrich_summary():
    print("--- Demonstrating Summary Enrichment (Mock) ---")
    
    # Ensure mock directories exist
    os.makedirs(MOCK_NWB_PATH, exist_ok=True)
    os.makedirs(MOCK_ARRAYS_PATH, exist_ok=True)
    
    # 1. Create a dummy DATA_AVAILABILITY_SUMMARY.md file
    mock_summary_path = os.path.join(MOCK_NWB_DIR, 'DATA_AVAILABILITY_SUMMARY.md')
    initial_content = """# 📊 Data Availability Summary (Mock Initial)

| Session | Trials | Conditions | Units (SPK) | LFP | Eye (x,y) | Pupil | Reward |
|:--------|:-------|:-----------|:------------|:----|:----------|:------|:-------|
| 230629  | 100    | 4          | 50          | Yes | Yes       | Yes   | Yes    |
| 230630  | 120    | 4          | 75          | Yes | Yes       | No    | Yes    |
"""
    with open(mock_summary_path, 'w', encoding='utf-8') as f:
        f.write(initial_content)
    print(f"  Mock initial summary created at: {mock_summary_path}")

    # 2. Simulate NWB files to get session IDs
    mock_nwb_filenames = ["sub-C31o_ses-230629_rec.nwb", "sub-C31o_ses-230630_rec.nwb"]
    for fn in mock_nwb_filenames:
        with open(os.path.join(MOCK_NWB_PATH, fn), 'w') as f:
            f.write("mock nwb content")
    
    # 3. Simulate .npy data files for one session
    session_id = "230629"
    cond = "AAAB"
    
    # Behavioral
    with open(os.path.join(MOCK_ARRAYS_PATH, f'ses{session_id}-behavioral-{cond}.npy'), 'w') as f: f.write("mock")
    # LFP
    for p in [0, 1]:
        with open(os.path.join(MOCK_ARRAYS_PATH, f'ses{session_id}-probe{p}-lfp-{cond}.npy'), 'w') as f: f.write("mock")
    # Units
    for p in [0]:
        with open(os.path.join(MOCK_ARRAYS_PATH, f'ses{session_id}-units-probe{p}-spk-{cond}.npy'), 'w') as f: f.write("mock")

    # --- Replicate core logic of enrich_summary function ---
    nwb_files = sorted([f for f in os.listdir(MOCK_NWB_PATH) if f.endswith('.nwb')])
    sessions = [f.split('_')[1].split('-')[1] for f in nwb_files]
    
    shape_data = []
    
    for sid in sessions:
        # We'll check for AAAB as a representative condition for shapes
        cond_check = "AAAB" # Use a generic condition for checking existence
        
        # Behavioral Shape
        behav_path = os.path.join(MOCK_ARRAYS_PATH, f'ses{sid}-behavioral-{cond_check}.npy')
        behav_shape = mock_get_npy_shape(behav_path) if os.path.exists(behav_path) else None
        
        # LFP Shapes (check P0, P1, P2)
        lfp_shapes = []
        for p in [0, 1, 2]:
            path = os.path.join(MOCK_ARRAYS_PATH, f'ses{sid}-probe{p}-lfp-{cond_check}.npy')
            shape = mock_get_npy_shape(path) if os.path.exists(path) else None
            if shape: lfp_shapes.append(f"P{p}: {shape}")
        
        # Unit Shapes (check P0, P1, P2)
        unit_shapes = []
        for p in [0, 1, 2]:
            path = os.path.join(MOCK_ARRAYS_PATH, f'ses{sid}-units-probe{p}-spk-{cond_check}.npy')
            shape = mock_get_npy_shape(path) if os.path.exists(path) else None
            if shape: unit_shapes.append(f"P{p}: {shape}")
            
        shape_data.append({
            "Session": sid,
            "Behavioral [T, 4, S]": str(behav_shape) if behav_shape else "⏳",
            "LFP Probes [T, 128, S]": ", ".join(lfp_shapes) if lfp_shapes else "⏳",
            "Unit Probes [T, N, S]": ", ".join(unit_shapes) if unit_shapes else "⏳"
        })

    df_shapes = pd.DataFrame(shape_data)
    
    with open(mock_summary_path, 'r', encoding='utf-8') as f:
        lines = f.readlines()
        
    new_lines = []
    skip = False
    for line in lines:
        if "### 📐 Detailed Array Shapes" in line:
            skip = True
        if skip and line.startswith("###") and "### 📐 Detailed Array Shapes" not in line:
            skip = False
        if not skip:
            new_lines.append(line)
            
    content = "".join(new_lines).strip()
    content += "

### 📐 Detailed Array Shapes
"
    content += "Detailed dimensional mapping for each session and probe. Format: `(Trials, Channels/Units, Samples)`.

"
    content += df_shapes.to_markdown(index=False)
    content += "

---
*Last updated by Gemini CLI Auto-Summary Suite.*"
    
    with open(mock_summary_path, 'w', encoding='utf-8') as f:
        f.write(content)
    
    print(f"
  Summary enriched with shapes at {mock_summary_path}")
    
    # Read the updated content to show the result
    with open(mock_summary_path, 'r', encoding='utf-8') as f:
        print("
--- Updated DATA_AVAILABILITY_SUMMARY.md Content ---")
        print(f.read())
    
    # Clean up mock directories and files
    import shutil
    shutil.rmtree(MOCK_NWB_DIR)
    print("
  Cleaned up mock environment.")

# --- Run the demonstration ---
if __name__ == '__main__':
    mock_enrich_summary()
```
